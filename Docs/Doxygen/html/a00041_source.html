126bool EVE_CoCmd_bitmapTransform(EVE_HalContext *phost, int32_t x0, int32_t y0, int32_t x1, int32_t y1, int32_t x2, int32_t y2, int32_t tx0, int32_t ty0, int32_t tx1, int32_t ty1, int32_t tx2, int32_t ty2, uint16_t *result)
127{
128 uint32_t resAddr;
129
130#if EVE_CMD_HOOKS
131 if (phost->CoCmdHook && phost->CoCmdHook(phost, CMD_BITMAP_TRANSFORM, 0))
132 return false;
133#endif
134
135 EVE_Cmd_startFunc(phost);
136 EVE_Cmd_wr32(phost, CMD_BITMAP_TRANSFORM);
137 EVE_Cmd_wr32(phost, x0);
138 EVE_Cmd_wr32(phost, y0);
139 EVE_Cmd_wr32(phost, x1);
140 EVE_Cmd_wr32(phost, y1);
141 EVE_Cmd_wr32(phost, x2);
142 EVE_Cmd_wr32(phost, y2);
143 EVE_Cmd_wr32(phost, tx0);
144 EVE_Cmd_wr32(phost, ty0);
145 EVE_Cmd_wr32(phost, tx1);
146 EVE_Cmd_wr32(phost, ty1);
147 EVE_Cmd_wr32(phost, tx2);
148 EVE_Cmd_wr32(phost, ty2);
149 resAddr = EVE_Cmd_moveWp(phost, 4);
150 EVE_Cmd_endFunc(phost);
151
152 /* Read result */
153 if (result)
154 {
155 if (!EVE_Cmd_waitFlush(phost))
156 return false;
157 *result = (uint16_t)EVE_Hal_rd32(phost, RAM_CMD + resAddr);
158 }
159 return true;
160}
