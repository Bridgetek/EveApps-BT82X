100/* I2C Master Controller Status
101 * bit 0 = controller busy: all other status bits invalid
102 * bit 1 = error condition
103 * bit 2 = slave address was not acknowledged during last operation
104 * bit 3 = data not acknowledged during last operation
105 * bit 4 = arbitration lost during last operation
106 * bit 5 = controller idle
107 * bit 6 = bus busy
108 */
109#define I2CM_CONTROLLER_BUSY(status) (((status) & 0x01) != 0)
110#define I2CM_DATA_NACK(status) (((status) & 0x0A) != 0)
111#define I2CM_ADDRESS_NACK(status) (((status) & 0x06) != 0)
112#define I2CM_ARB_LOST(status) (((status) & 0x12) != 0)
113#define I2CM_IDLE(status) (((status) & 0x20) != 0)
114#define I2CM_BUS_BUSY(status) (((status) & 0x40) != 0)
