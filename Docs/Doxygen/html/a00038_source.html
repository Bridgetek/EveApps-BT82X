152uint32_t EVE_CoCmd_flashAttach(EVE_HalContext *phost)
153{
154 uint32_t flashStatus;
155 if (!EVE_Cmd_waitFlush(phost))
156 return EVE_Hal_rd32(phost, REG_FLASH_STATUS); // Coprocessor must be ready
157 flashStatus = EVE_Hal_rd32(phost, REG_FLASH_STATUS);
158 if (flashStatus != FLASH_STATUS_DETACHED)
159 return flashStatus; // Only attach when detached
160 EVE_CoCmd_d(phost, CMD_FLASHATTACH);
161 EVE_Cmd_waitFlush(phost); // Wait for command completion
162 return EVE_Hal_rd32(phost, REG_FLASH_STATUS); // Return current status
163}

165uint32_t EVE_CoCmd_flashFast(EVE_HalContext *phost, uint32_t *result)
166{
167 uint32_t resAddr;
168 uint32_t flashStatus;
169
170 if (!EVE_Cmd_waitFlush(phost))
171 {
172 if (result)
173 *result = 0xE000;
174 return EVE_Hal_rd32(phost, REG_FLASH_STATUS);
175 } // Coprocessor must be ready
176 flashStatus = EVE_Hal_rd32(phost, REG_FLASH_STATUS);
177 if (flashStatus < FLASH_STATUS_BASIC)
178 {
179 if (result)
180 *result = 0xE001;
181 return flashStatus;
182 }
183 if (flashStatus > FLASH_STATUS_BASIC)
184 {
185 if (result)
186 *result = 0;
187 return flashStatus;
188 } // Only enter fast mode when attached
189
190#if EVE_CMD_HOOKS
191 if (phost->CoCmdHook && phost->CoCmdHook(phost, CMD_FLASHFAST, 0))
192 return flashStatus;
193#endif
194
195 EVE_Cmd_startFunc(phost);
196 EVE_Cmd_wr32(phost, CMD_FLASHFAST);
197 resAddr = EVE_Cmd_moveWp(phost, 4); // Get the address where the coprocessor will write the result
198 EVE_Cmd_endFunc(phost);
199 EVE_Cmd_waitFlush(phost); // Wait for command completion
200 if (result)
201 *result = EVE_Hal_rd32(phost, RAM_CMD + resAddr); // Fetch result
202 return EVE_Hal_rd32(phost, REG_FLASH_STATUS); // Return current status
203}

230bool EVE_CoCmd_loadImage_flash(EVE_HalContext *phost, uint32_t dst, uint32_t src, uint32_t *format)
231{
232#if EVE_CMD_HOOKS
233 if (phost->CoCmdHook && phost->CoCmdHook(phost, CMD_FLASHSOURCE, src))
234 return false;
235 if (phost->CoCmdHook && phost->CoCmdHook(phost, CMD_LOADIMAGE, dst))
236 return false;
237#endif
238
239 if (!EVE_Cmd_waitFlush(phost))
240 return false; // Coprocessor must be ready
241 EVE_Cmd_startFunc(phost);
242 EVE_Cmd_wr32(phost, CMD_FLASHSOURCE);
243 EVE_Cmd_wr32(phost, src);
244 EVE_Cmd_wr32(phost, CMD_LOADIMAGE);
245 EVE_Cmd_wr32(phost, dst);
246 EVE_Cmd_wr32(phost, OPT_FLASH | OPT_NODL);
247 EVE_Cmd_endFunc(phost);
248 if (!EVE_Cmd_waitFlush(phost))
249 return false; // Image failed to load
250 if (format)
251 *format = EVE_Hal_rd32(phost, 0x3097e8);
252 return true;
