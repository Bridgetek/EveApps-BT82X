541 return 0;
542 }
543 if (!phost->LoadFileHandle)
544 {

547 fseek(afile, transfered ? *transfered : 0, SEEK_SET);

631static bool loadMediaFile(EVE_HalContext *phost, const char *filename, const wchar_t *filenameW, uint32_t *transfered)
632#else
633bool EVE_Util_loadMediaFile(EVE_HalContext *phost, const char *filename, uint32_t *transfered)
634#endif
635{
636 FILE *afile;
637 uint32_t remaining = 0;
638 uint16_t blockSize = (uint16_t)((phost->MediaFifoSize >> 3) << 2) - 4;
639#pragma warning(push)
640#pragma warning(disable : 6255)
641 uint8_t *pbuff = (uint8_t *)_alloca(blockSize);
642#pragma warning(pop)
643 uint16_t blocklen;
644 errno_t err = 0;
645 if (!transfered)
646 EVE_Util_closeFile(phost);
647 if (phost->CmdFault)
648 return false;
649 if (phost->LoadFileHandle)
650 {
651 afile = phost->LoadFileHandle;
652 }
653#ifdef _WIN32
654 else if (!filename)
655 {
656 err = _wfopen_s(&afile, filenameW, L"rb");
657 }
658#endif
659 else
660 {
661 err = fopen_s(&afile, filename, "rb");
662 }
663 if (err || afile == NULL)
664 {
665#ifdef _WIN32
666 if (!filename)
667 eve_printf_debug("Unable to open: %ls\n", filenameW);
668 else
669#endif
670 eve_printf_debug("Unable to open: %s\n", filename);
671 return false;
672 }
673 if (!phost->LoadFileHandle)
674 {
675 fseek(afile, 0, SEEK_END);
676 remaining = ftell(afile);
677 fseek(afile, transfered ? *transfered : 0, SEEK_SET);
678 if (transfered)
679 phost->LoadFileHandle = afile;
680 }
681 else
682 {
683 remaining = (uint32_t)phost->LoadFileRemaining;
684 }
685 while (remaining > 0)
686 {
687 blocklen = (uint16_t)min(blockSize, remaining);
688 blocklen = (uint16_t)fread((void *)pbuff, 1, blocklen, afile); /* Copy the data into pbuff and then transfter it to command buffer */
689
690 if (blocklen == 0)
691 {
692 eve_printf_debug("Read 0 bytes, unexpected end of file, %i bytes remaining\n", (int)remaining);
693 break;
694 }
695
696 remaining -= blocklen;
697 blocklen += 3;
698 blocklen &= ~3U;
699
700 if (transfered)
701 {
702 uint32_t transferedPart = 0;
703 bool wrRes = EVE_MediaFifo_wrMem(phost, pbuff, blocklen, &transferedPart); /* Copy data continuously into media fifo memory */
704 *transfered += transferedPart;
705 if (transferedPart < blocklen)
706 {
707 long offset = (long)transferedPart - (long)blocklen; /* Negative */
708 fseek(afile, offset, SEEK_CUR); /* Seek back */
709 remaining -= offset; /* Increments remaining (double negative) */
710 break; /* Early exit, processing done */
711 }
712 if (!wrRes)
713 break;
714 }
715 else
716 {
717 if (!EVE_MediaFifo_wrMem(phost, pbuff, blocklen, NULL)) /* Copy data continuously into media fifo memory */
718 break; /* Coprocessor fault */
719 }
720 }
721 if (!transfered)
722 {
723 fclose(afile); /* Close the opened file */
724 }
725 else if (remaining)
726 {
727 phost->LoadFileRemaining = remaining; /* Save remaining */
728 }
729 else
730 {
731 EVE_Util_closeFile(phost);
732 }
733 return transfered ? EVE_Cmd_waitFlush(phost) : EVE_MediaFifo_waitFlush(phost, false);
734}
