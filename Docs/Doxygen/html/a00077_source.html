105bool EVE_HalImpl_defaults(EVE_HalParameters *parameters, size_t deviceIdx)
106{
107 FT_DEVICE_LIST_INFO_NODE chanInfo = { 0 };
108 bool res = deviceIdx >= 0 && deviceIdx < s_NumChannels;
109 if (!res)
110 {
111 uint32_t i;
112 if (!s_NumChannels)
113 EVE_Hal_list();
114
115 // Select first open device
116 deviceIdx = 0;
117 for (i = 0; i < s_NumChannels; ++i)
118 {
119 if (SPI_GetChannelInfo((uint32_t)i, &chanInfo) != FT_OK)
120 continue;
121 if (!(chanInfo.Flags & FT_FLAGS_OPENED))
122 {
123 deviceIdx = i;
124 res = true;
125 break;
126 }
127 }
128 }
129 else
130 {
131 SPI_GetChannelInfo((uint32_t)deviceIdx, &chanInfo);
132 }
133 parameters->MpsseChannelNo = deviceIdx & 0xFF;
134 parameters->PowerDownPin = 7;
135 parameters->SpiClockrateKHz = 15000; /* in KHz */
136 if (!strcmp(chanInfo.Description, "UMFTPD2A B"))
137 {
138 /* Settings for FT9xx GPIO passthrough */
139 /* parameters->PowerDownPin = 7 + 16; */ /* Uses channel D, useful for FT9xx passthrough. See EVE_HalImpl_open */
140 /* parameters->SpiClockrateKHz = 600; */
141
142 /* Settings for FT9xx hardware SPI passthrough */
143 parameters->PowerDownPin = 0x80 + 43; /* GPIO pin 43 */
144 parameters->SpiClockrateKHz = 2000; // 2000;
145 }
146 return res;

157bool EVE_HalImpl_open(EVE_HalContext *phost, const EVE_HalParameters *parameters)
158{
159 FT_STATUS status;
160 FT_DEVICE_LIST_INFO_NODE chanInfo = { 0 };
161 ChannelConfig channelConf; /* channel configuration */
162
163 /* configure the spi settings */
164 channelConf.ClockRate = parameters->SpiClockrateKHz * 1000;
165 channelConf.LatencyTimer = 2;
166 channelConf.configOptions = SPI_CONFIG_OPTION_MODE0 | SPI_CONFIG_OPTION_CS_DBUS3 | SPI_CONFIG_OPTION_CS_ACTIVELOW;
167 channelConf.Pin = 0x00008080; /* FinalVal-FinalDir-InitVal-InitDir (for dir 0=in, 1=out) */ /* !!! Critical setting for system to power on */
168
169 phost->SpiClockrateKHz = parameters->SpiClockrateKHz;
170 phost->MpsseChannelNo = parameters->MpsseChannelNo;
171 phost->PowerDownPin = parameters->PowerDownPin;
172
173 /* Open the first available channel */
174 status = SPI_OpenChannel(parameters->MpsseChannelNo, (FT_HANDLE *)&phost->SpiHandle);
175 phost->GpioHandle = phost->SpiHandle;
176 if (FT_OK != status)
177 {
178 eve_printf_debug("SPI open channel failed %d %p\n", parameters->MpsseChannelNo, phost->SpiHandle);
179 return false;
180 }
181 status = SPI_InitChannel((FT_HANDLE)phost->SpiHandle, &channelConf);
182 if (FT_OK != status)
183 {
184 eve_printf_debug("SPI init channel failed %d %p\n", parameters->MpsseChannelNo, phost->SpiHandle);
185 return false;
186 }
187
188 eve_printf_debug("\nhandle=0x%p status=0x%x\n", phost->SpiHandle, (unsigned int)status);
189
190 /* Special case, when connecting through UMFTPD2A, use channel D for GPIO */
191 if (phost->PowerDownPin >= 8 && ((phost->PowerDownPin & 0x80) == 0))
192 {
193 SPI_GetChannelInfo((uint32_t)phost->MpsseChannelNo, &chanInfo);
194 if (!strcmp(chanInfo.Description, "UMFTPD2A B"))
195 {
196 size_t slen = strlen(chanInfo.SerialNumber);
197 char cn = 'B' + (phost->PowerDownPin >> 3);
198 eve_printf_debug("%s (%s): Adding channel %c for GPIO\n", chanInfo.Description, chanInfo.SerialNumber, cn);
199 phost->PowerDownPin &= 7;
200 chanInfo.SerialNumber[slen - 1] = cn;
201 phost->GpioHandle = NULL;
202 if (FT_OpenEx(chanInfo.SerialNumber, FT_OPEN_BY_SERIAL_NUMBER, &phost->GpioHandle) != FT_OK)
203 {
204 eve_printf_debug("Failed to open channel D\n");
205 FT_Close(phost->SpiHandle);
206 return false;
207 }
208 else
209 {
210 eve_printf_debug("Channel D open OK\n");
211 }
212 if (FT_ResetDevice(phost->GpioHandle) != FT_OK
213 || FT_SetBitMode(phost->GpioHandle, (1 << phost->PowerDownPin), FT_BITMODE_SYNC_BITBANG) != FT_OK)
214 {
215 eve_printf_debug("Failed to prepare channel D\n");
216 FT_Close(phost->GpioHandle);
217 FT_Close(phost->SpiHandle);
218 return false;
219 }
220 }
221 }
222
223 /* Initialize the context variables */
224 phost->SpiChannels = EVE_SPI_SINGLE_CHANNEL;
225 phost->Status = EVE_STATUS_OPENED;
226 ++g_HalPlatform.OpenedDevices;
227
228#ifdef _WIN32
229 /* Performance degradation workaround. Mitigate SPI_ToggleCS sleep, if it's detected. */
230 /* SPI_ToggleCS may be fixed to remove the sleep call in future versions of libMPSSE. */
231 /* This workaround applies only if the affected libMPSSE library is being used. */
232 if (!s_SleepMitigated)
233 {
234 /* This will sleep twice, once on each FALSE */
235 uint32_t ms, delta;
236 ms = EVE_millis();
237 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, TRUE);
238 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, FALSE);
239 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, TRUE);
240 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, FALSE);
241 delta = EVE_millis() - ms;
242 /* If the sleep time is more than a single full tick (either 15ms or 16ms),
243 we know for sure that it's sleeping a full tick twice. */
244 if (delta > 16)
245 {
246 /* Set thread scheduler for process to 1ms. */
247 timeBeginPeriod(1);
248 s_SleepMitigated = true;
249 /* Verify that it worked. */
250 eve_printf_debug("Mitigating libMPSSE sleep issue (%u ms)\n", (unsigned int)delta);
251 ms = EVE_millis();
252 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, TRUE);
253 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, FALSE);
254 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, TRUE);
255 SPI_ToggleCS((FT_HANDLE)phost->SpiHandle, FALSE);
256 delta = EVE_millis() - ms;
257 /* EVE_millis may jump a whole tick at most */
258 if (delta > 16)
259 {
260 eve_printf_debug("MPSSE sleep issue may still persist (%u ms)\n", (unsigned int)delta);
261 }
262 else
263 {
264 eve_printf_debug("MPSSE sleep issue mitigated\n");
265 }
266 }
267 }
268#endif
269
270 return true;
271}

319static inline uint32_t incrementRamGAddr(EVE_HalContext *phost, uint32_t addr, uint32_t inc)
320{
321 if (addr != REG_CMDB_WRITE)
322 {
323 bool wrapCmdAddr = (addr >= RAM_CMD) && (addr < (RAM_CMD + EVE_CMD_FIFO_SIZE));
324 addr += inc;
325 if (wrapCmdAddr)
326 addr = RAM_CMD + (addr & EVE_CMD_FIFO_MASK);
327 }
328 return addr;

340static inline bool rdBuffer(EVE_HalContext *phost, uint8_t *buffer, uint32_t size)
341{
342 uint32_t sizeRemaining = size;
343 if (sizeRemaining & 3)
344 {
345 sizeRemaining = (sizeRemaining + 3) & ~3UL;
346 eve_printf_debug("Data size should be align to 4 bytes\n");
347 }
348
349 while (sizeRemaining)
350 {
351 uint32_t sizeTransferred = 0;
352 uint32_t bytesPerRead;
353 bool readyRecved = false;
354 uint8_t retry = 0;
355 uint8_t buff1[DATA_LENGTH + POLLING_BYTES] = { 0 };
356
357 if (sizeRemaining <= DATA_LENGTH)
358 bytesPerRead = sizeRemaining;
359 else
360 bytesPerRead = DATA_LENGTH;
361
362 while (1)
363 {
364 FT_STATUS status = SPI_Read(phost->SpiHandle, buff1, bytesPerRead + POLLING_BYTES, &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES);
365 if ((status != FT_OK) || (sizeTransferred != (POLLING_BYTES + bytesPerRead)))
366 {
367 eve_printf_debug("SPI_Read failed, sizeTransferred is %d with status %d\n", sizeTransferred, status);
368 if (sizeTransferred != (POLLING_BYTES + bytesPerRead))
369 phost->Status = EVE_STATUS_ERROR;
370 return false;
371 }
372
373 for (uint8_t i = 0; i < POLLING_BYTES; i++)
374 {
375 if (buff1[i] == 0x01)
376 {
377 readyRecved = true;
378 memcpy(buffer, &buff1[i + 1], bytesPerRead); /* Prepare the SFN to be modified */
379 sizeTransferred = bytesPerRead;
380 break;
381 }
382 }
383 if (readyRecved)
384 break;
385
386 retry++;
387 if (retry >= READ_TIMEOUT)
388 return false;
389 }
390 sizeRemaining -= sizeTransferred;
391 buffer += sizeTransferred;
392 }
393
394 return true;
395}

406static inline bool wrBuffer(EVE_HalContext *phost, const uint8_t *buffer, uint32_t size)
407{
408#if defined(EVE_BUFFER_WRITES)
409 if (size & 3)
410 {
411 size = (size + 3) & ~3UL;
412 eve_printf_debug("Data size should be align to 4 bytes\n");
413 }
414 if (buffer && (size < (sizeof(phost->SpiWrBuf) - phost->SpiWrBufIndex)))
415 {
416 /* Write to buffer */
417 memcpy(&phost->SpiWrBuf[phost->SpiWrBufIndex], buffer, size);
418 phost->SpiWrBufIndex += size;
419 return true;
420 }
421 else
422 {
423 if (buffer && phost->SpiWrBufIndex)
424 {
425 /* Buffer is over size, flush now */
426 if (!flush(phost))
427 return false;
428
429 /* Write to buffer */
430 if (size < sizeof(phost->SpiWrBuf))
431 return wrBuffer(phost, buffer, size);
432 }
433
434 if (buffer || phost->SpiWrBufIndex)
435 {
436 uint32_t sizeTransferred;
437 uint8_t hrdpkt[8];
438 uint32_t addr = phost->SpiRamGAddr;
439 FT_STATUS status;
440 uint32_t sizeRemaining;
441
442 if (!buffer)
443 {
444 /* Flushing */
445 buffer = phost->SpiWrBuf;
446 size = phost->SpiWrBufIndex;
447 phost->SpiWrBufIndex = 0;
448 }
449
450 /* Compose the HOST MEMORY WRITE packet */
451 hrdpkt[0] = (addr >> 24) | 0x80; /* MSB bit 1 for WRITE */
452 hrdpkt[1] = (addr >> 16) & 0xFF;
453 hrdpkt[2] = (addr >> 8) & 0xFF;
454 hrdpkt[3] = addr & 0xFF;
455
456 status = SPI_Write((FT_HANDLE)phost->SpiHandle, hrdpkt, 4, &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES | SPI_TRANSFER_OPTIONS_CHIPSELECT_ENABLE);
457
458 if ((status != FT_OK) || (sizeTransferred != 4))
459 {
460 eve_printf_debug("%d SPI_Write failed, sizeTransferred is %d with status %d\n", __LINE__, sizeTransferred, (int)status);
461 if (sizeTransferred != 4)
462 phost->Status = EVE_STATUS_ERROR;
463 return false;
464 }
465
466 sizeRemaining = size;
467 while (sizeRemaining)
468 {
469 uint32_t transferSize = min(0xFFFF, sizeRemaining);
470 FT_STATUS status = SPI_Write(phost->SpiHandle, (uint8 *)buffer, transferSize, &sizeTransferred,
471 (transferSize == sizeRemaining) ? (SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES | SPI_TRANSFER_OPTIONS_CHIPSELECT_DISABLE) : SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES);
472 sizeRemaining -= sizeTransferred;
473 buffer += sizeTransferred;
474
475 if (status != FT_OK || !sizeTransferred)
476 {
477 eve_printf_debug("%d SPI_Write failed, sizeTransferred is %d with status %d\n", __LINE__, sizeTransferred, (int)status);
478 phost->Status = EVE_STATUS_ERROR;
479 return false;
480 }
481 }
482
483 addr = incrementRamGAddr(phost, addr, size);
484 phost->SpiRamGAddr = addr;
485 }
486
487 return true;
488 }
489#else
490 uint32_t sizeTransferred = 0;
491 uint32_t sizeRemaining = size;
492
493 while (sizeRemaining)
494 {
495 FT_STATUS status = SPI_Write(phost->SpiHandle, (uint8 *)buffer, min(0xFFFF, sizeRemaining), &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES);
496 sizeRemaining -= sizeTransferred;
497 buffer += sizeTransferred;
498
499 if (status != FT_OK || !sizeTransferred)
500 {
501 eve_printf_debug("%d SPI_Write failed, sizeTransferred is %d with status %d\n", __LINE__, sizeTransferred, status);
502 phost->Status = EVE_STATUS_ERROR;
503 return false;
504 }
505 }
506
507 return true;
508#endif
509}

551void EVE_Hal_startTransfer(EVE_HalContext *phost, EVE_TRANSFER_T rw, uint32_t addr)
552{
553 eve_assert(phost->Status == EVE_STATUS_OPENED);
554
555#if defined(EVE_BUFFER_WRITES)
556 if (addr != incrementRamGAddr(phost, phost->SpiRamGAddr, phost->SpiWrBufIndex) || rw == EVE_TRANSFER_READ)
557 {
558 /* Close any write transfer that was left open, if the address changed */
559 flush(phost);
560 phost->SpiRamGAddr = addr;
561 }
562#endif
563
564 if (rw == EVE_TRANSFER_READ)
565 {
566 uint8_t transferArray[4];
567 uint32_t sizeTransferred;
568
569#if defined(EVE_BUFFER_WRITES)
570 eve_assert(!phost->SpiWrBufIndex);
571#endif
572
573 /* Compose the read packet */
574 transferArray[0] = (addr >> 24) & 0xFF;
575 transferArray[1] = (addr >> 16) & 0xFF;
576 transferArray[2] = (addr >> 8) & 0xFF;
577 transferArray[3] = addr & 0xFF;
578
579 SPI_Write((FT_HANDLE)phost->SpiHandle, transferArray, 4, &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES | SPI_TRANSFER_OPTIONS_CHIPSELECT_ENABLE);
580
581 if (phost->Status != EVE_STATUS_ERROR)
582 phost->Status = EVE_STATUS_READING;
583 }
584 else
585 {
586#if defined(EVE_BUFFER_WRITES)
587 if (phost->Status != EVE_STATUS_ERROR)
588 phost->Status = EVE_STATUS_WRITING;
589#else
590 uint8_t transferArray[4];
591 uint32_t sizeTransferred;
592
593 /* Compose the write packet */
594 transferArray[0] = (addr >> 24) | 0x80;
595 transferArray[1] = (addr >> 16) & 0xFF;
596 transferArray[2] = (addr >> 8) & 0xFF;
597 transferArray[3] = addr & 0xFF;
598
599 SPI_Write((FT_HANDLE)phost->SpiHandle, transferArray, 4, &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES | SPI_TRANSFER_OPTIONS_CHIPSELECT_ENABLE);
600
601 if (phost->Status != EVE_STATUS_ERROR)
602 phost->Status = EVE_STATUS_WRITING;
603#endif
604 }

613{

643 phost->Status = EVE_STATUS_OPENED;

650 {

810static FT_STATUS EVE_HalImpl_passthroughGpio(EVE_HalContext *phost, uint8_t gpio, uint8_t val)
811{
812 /* Special behaviour for FT9xx passthrough to control GPIO pins on FT9xx.
813 This produces a fake read */
814
815 int tries = 0;
816
817 uint8_t recvArray[2];
818 uint8_t transferArray[8];
819 uint32_t sizeTransferred;
820 FT_STATUS ftRes;
821
822 transferArray[0] = 0;
823 transferArray[1] = 0;
824 transferArray[2] = 0;
825 transferArray[3] = 0;
826 transferArray[4] = 0xA5;
827 transferArray[5] = gpio;
828 transferArray[6] = val;
829 transferArray[7] = 0xA5 ^ gpio ^ val;
830
831 recvArray[1] = 0;
832
833#if defined(EVE_BUFFER_WRITES)
834 flush(phost);
835#endif
836 do
837 {
838 EVE_sleep(20);
839 ftRes = SPI_Write(phost->SpiHandle, transferArray, sizeof(transferArray), &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES | SPI_TRANSFER_OPTIONS_CHIPSELECT_ENABLE)
840 || SPI_Read(phost->SpiHandle, recvArray, sizeof(recvArray), &sizeTransferred, SPI_TRANSFER_OPTIONS_SIZE_IN_BYTES | SPI_TRANSFER_OPTIONS_CHIPSELECT_DISABLE);
841 ++tries;
842 } while (ftRes == FT_OK && recvArray[1] != 0x5A && tries < 128);
843
844 if (tries >= 128 && ftRes == FT_OK)
845 ftRes = FT_IO_ERROR;
846
847 return ftRes;
848}

858bool EVE_Hal_powerCycle(EVE_HalContext *phost, bool up)
