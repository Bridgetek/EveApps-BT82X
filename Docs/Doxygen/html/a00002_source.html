92int32_t Gpu_Hal_Dec2Ascii(char *pSrc, int32_t value)
93{
94 int16_t Length;
95 char *pdst;
96 char charval;
97 int32_t CurrVal = value;
98 int32_t tmpval;
99 int32_t i;
100 char tmparray[16];
101 char idx = 0;
102
103 Length = (int16_t)strlen((char *)pSrc);
104 pdst = pSrc + Length;
105
106 if (0 == value)
107 {
108 *pdst++ = '0';
109 *pdst++ = '\0';
110 return 0;
111 }
112
113 if (CurrVal < 0)
114 {
115 *pdst++ = '-';
116 CurrVal = -CurrVal;
117 }
118 /* insert the value */
119 while (CurrVal > 0)
120 {
121 tmpval = CurrVal;
122 CurrVal /= 10;
123 tmpval = tmpval - CurrVal * 10;
124 charval = '0' + tmpval;
125 tmparray[idx++] = charval;
126 }
127
128 for (i = 0; i < idx; i++)
129 {
130 *pdst++ = tmparray[idx - i - 1];
131 }
132 *pdst++ = '\0';
133
134 return 0;
135}

157 eve_printf("result %lx\n", result);
158 EVE_Cmd_waitFlush(phost);
159
160 eve_printf_debug("App_CoPro_Widget_Calibrate: End\n");
161
162 // Print the configured values
163 EVE_Hal_rdMem(phost, (uint8_t *)transMatrix, REG_TOUCH_TRANSFORM_A, 4 * 6); // read all the 6 coefficients
164 eve_printf_debug("Touch screen transform values are A 0x%lx,B 0x%lx,C 0x%lx,D 0x%lx,E 0x%lx, F 0x%lx\n",

326void Flash_Init(EVE_HalContext* phost, const uint8_t *filePath, const uint8_t *fileName)
327{
328#if defined(_WIN32)
329#define _WHERE "PC"
330#endif
331
333 if (!Show_Diaglog_YesNo(phost, "Flash programming",
334 "Program the flash with the file in " _WHERE "?")) {
336 return;
337 }
338
340 int sent = Ftf_Write_File_To_Flash_With_Progressbar(phost, filePath,
341 fileName, 0);
342
344 if (0 >= sent) {
345 uint8_t count = 5;
346 uint8_t msg[1000];
347 while (count) {
348 snprintf(msg, 1000, "Error: Cannot open file: %s, exit in %u s",
349 fileName, count);
350 Show_Diaglog_Info(phost, msg);
351 EVE_sleep(1000);
352 count--;
353 }
354 exit(0);

672void LVDS_Config(EVE_HalContext *phost, uint16_t format, Display_mode mode)
673{
674 uint8_t TXPLLDiv = 0;
675 uint8_t lvdspll_cps = 0;
676 uint16_t lock_delay = 0x180; //384
677 uint8_t lvdspll_cks = 0;
678 uint8_t lvdspll_ns = 7;
679 uint16_t w, h;
680#if defined(DISPLAY_RESOLUTION_WUXGA)
681 w = 1920;
682 h = 1200;
683 TXPLLDiv = 0x03;
684 eve_printf_debug("TXPLLDiv %d\n", TXPLLDiv);
685 lvdspll_cks = TXPLLDiv > 4 ? 1 : 2;
686 EVE_Hal_wr32(phost, REG_LVDSTX_PLLCFG, LVDSTX_PLLCFG(lvdspll_cps, lock_delay, lvdspll_cks, lvdspll_ns, TXPLLDiv));
687#endif
688 EVE_Hal_wr32(phost, REG_LVDSTX_EN, 0);
689 EVE_Hal_wr32(phost, REG_LVDSTX_EN, LVDS_CH1_EN | LVDS_CH0_EN);
690 EVE_sleep(10);
691 eve_printf_debug("LVDSTX_EN: %lx \n", EVE_Hal_rd32(phost, REG_LVDSTX_EN));
692 eve_printf_debug("LVDSTX_PLLCFG: %lx \n", EVE_Hal_rd32(phost, REG_LVDSTX_PLLCFG));
693 eve_printf_debug("LVDSTX_CTRL_CH0: %lx \n", EVE_Hal_rd32(phost, REG_LVDSTX_CTRL_CH0));
694 eve_printf_debug("LVDSTX_CTRL_CH1: %lx \n", EVE_Hal_rd32(phost, REG_LVDSTX_CTRL_CH1));
695
696 EVE_Hal_wr32(phost, REG_SC0_SIZE, 2); // set number of buffers to 2
697 EVE_Hal_wr32(phost, REG_SC0_PTR0, SC0_PTR0_STARTADDR);
698 EVE_Hal_wr32(phost, REG_SC0_PTR1, SC0_PTR1_STARTADDR);
699
700 EVE_Hal_wr32(phost, REG_SC1_SIZE, 2); // set number of buffers to 2
701 EVE_Hal_wr32(phost, REG_SC1_PTR0, SC1_PTR0_STARTADDR);
702 EVE_Hal_wr32(phost, REG_SC1_PTR1, SC1_PTR1_STARTADDR);
703
704 EVE_Hal_wr32(phost, REG_SC2_SIZE, 2); // set number of buffers to 2
705 EVE_Hal_wr32(phost, REG_SC2_PTR0, SC2_PTR0_STARTADDR);
706 EVE_Hal_wr32(phost, REG_SC2_PTR1, SC2_PTR1_STARTADDR);
707 EVE_Cmd_waitFlush(phost);
708
709 EVE_Hal_wr32(phost, REG_SO_MODE, FOUR_PIXEL_DUAL_LVDS);
710 if (mode == MODE_PICTURE)
711 {
712 scanout_single(phost, format, w, h);
713 }
714 else if (mode == MODE_DIRECTVIDEO)
715 {
716 EVE_Hal_wr32(phost, REG_SO_FORMAT, format);
717
718 EVE_Hal_wr32(phost, REG_SO_EN, 0); // disable
719 EVE_Hal_wr32(phost, REG_SC1_RESET, 1); // write to reset
720 EVE_Hal_wr32(phost, REG_SO_SOURCE, SWAPCHAIN_1);
721 EVE_Hal_wr32(phost, REG_SO_EN, 1); // enable
722 }
723 else if (mode == MODE_VIDEO)
724 {
725 scanout_swapping(phost, format, w, h);
726 }
727 else if (mode == MODE_LVDSRX)
728 {
729 EVE_Hal_wr32(phost, REG_LVDSRX_SETUP, ((LVDS_MODE_VESA_24 << 3) | (VS_POL_HIGH << 2) | (LVDSRX_TWO_CHANNEL) << 1 | LVDSRX_ONE_PIXEL_PER_CLK));
730 EVE_Hal_wr32(phost, REG_LVDSRX_CTRL, ((8 << 12) | (CHn_CLKSEL_FALLING << 11) | (CHn_FRANGE_10_30 << 9) | (CHn_PWDN_B_ON << 8) |
731 (8 << 4) | (CHn_CLKSEL_FALLING << 3) | (CHn_FRANGE_10_30 << 1) | CHn_PWDN_B_ON));
732
733 EVE_Hal_wr32(phost, REG_LVDSRX_CORE_ENABLE, 1);
734 EVE_Hal_wr32(phost, REG_LVDSRX_CORE_CAPTURE, 1);
735 EVE_Hal_wr32(phost, REG_LVDSRX_CORE_SETUP, ((LVDSRX_TWO_CHANNEL << 1) | (LVDSRX_ONE_PIXEL_PER_CLK & 0x1)));
736 EVE_Hal_wr32(phost, REG_LVDSRX_CORE_DEST, SWAPCHAIN_2);
737 EVE_Hal_wr32(phost, REG_LVDSRX_CORE_FORMAT, RGB8);
738 EVE_Hal_wr32(phost, REG_LVDSRX_CORE_DITHER, 0);
739
740 scanout_swapping(phost, format, w, h);
741 }
742 else
743 {
744 scanout_swapping(phost, format, w, h);
745 }
746
747 EVE_Cmd_waitFlush(phost);
748 eve_printf_debug("LVDSTX_STAT is %lx \n", EVE_Hal_rd32(phost, REG_LVDSTX_STAT));
749 eve_printf_debug("LVDSTX_ERR_STAT is %lx \n", EVE_Hal_rd32(phost, REG_LVDSTX_ERR_STAT));
750}
