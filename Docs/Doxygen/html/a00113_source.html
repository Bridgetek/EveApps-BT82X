125static bool configDefaultsEx(EVE_HalContext *phost, EVE_ConfigParameters *config, uint32_t width, uint32_t height, uint32_t refreshRate, uint32_t hsfWidth, uint32_t freq)
126{
127 /* Only false if the output resolution will be wrong.
128 Not affected by unsupported HSF or refreshRate */
129 bool supportedResolution = true;
130
131 uint32_t screenWidth;
132 uint32_t pixels;
133 uint32_t minCycles;
134 uint32_t maxRate;
135 uint32_t pclk;
136
137 uint32_t cycles;
138 uint32_t vcycle;
139 uint32_t hcycle;
140 uint32_t hoffset;
141 uint32_t voffset;
142 uint32_t hsync1;
143 uint32_t vsync1;
144
145 memset(config, 0, sizeof(EVE_ConfigParameters));
146
147 /*
148 Refresh rate is as follows:
149 REG_VCYCLE * REG_HCYCLE * REG_PCLK is the number of clocks each frame takes,
150 Divide REG_FREQUENCY by this number to get the frame refresh rate in Hz.
151 REG_HCYCLE must be higher than REG_HSIZE, REG_VCYCLE must be higher than REG_VSIZE.
152
153 By default, BT820 is at 72MHz.
154 User may configure this differently in `EVE_Util_bootup`.
155
156 Clocks per frame by clock frequency at 60Hz:
157 - 84MHz: 1400k
158 - 72MHz: 1200k
159 - 60MHz: 1000k
160 - 48MHz: 800k
161 - 36MHz: 600k
162 - 24MHz: 400k
163 */
164
165 /* Trim oversize displays */
166 if (width > 2047)
167 {
168 eve_printf_debug("Line buffer width %i is too large for this configuration\n", (int)config->Width);
169 width = 2047;
170 supportedResolution = false;
171 }
172 if (height > 2047)
173 {
174 eve_printf_debug("Screen height %i is too large for this configuration\n", (int)config->Height);
175 height = 2047;
176 supportedResolution = false;
177 }
178
179 if (hsfWidth)
180 {
181 /* Use the screen width if HSF is not supported */
182 config->Width = (int16_t)hsfWidth;
183 eve_printf_debug("Display config specifies HSF but not supported on this platform\n");
184 }
185 else
186 {
187 config->Width = (int16_t)width;
188 }
189 config->Height = (int16_t)height;
190
191 screenWidth = hsfWidth ? hsfWidth : width; /* Use screen width for calculation */
192 pixels = screenWidth * height;
193
194 /* Calculate maximum refresh rate */
195 minCycles = pixels + (pixels >> 2); /* pixels * 1.25 */
196 maxRate = freq / minCycles;
197
198 /* If the refresh rate is too low, try with tighter settings */
199 if (refreshRate > maxRate)
200 {
201 minCycles = pixels + (pixels >> 3); /* pixels * 1.125 */
202 maxRate = freq / minCycles;
203 }
204
205 if (maxRate < refreshRate)
206 {
207 /* Trim unsupported framerate */
208 eve_printf_debug("Frame rate limited to %d\n", (unsigned int)maxRate);
209 refreshRate = maxRate;
210 pclk = 1;
211 }
212 else
213 {
214 pclk = maxRate / refreshRate;
215 }
216 config->PCLK = (uint8_t)pclk;
217
218 /* Change PCLKPol if the right edge of the display is noisy,
219 or if the display is showing jagged colors. Values 0 or 1.
220 This toggles the polarity of the PCLK. */
221 config->PCLKPol = 1; /* non-default */
222
223 /* Approximate an average good setting */
224 cycles = freq / (refreshRate * pclk);
225 vcycle = height * cycles / pixels; /* max vcycle without blanking, on average 1.125x height, or 1.25x */
226 vcycle = (vcycle + ((vcycle + height) >> 1) + height + height) >> 2; /* vcycle is 1.5 max vcycle + 2.5 height / 4, so on average 1.046875x height, or 1.09375x */
227 hcycle = cycles / vcycle; /* cycles div by vcycle, on average 1.07375x, or 1.1425x */
228 hoffset = (hcycle - screenWidth) >> 1;
229 voffset = (vcycle - height) >> 1;
230 hsync1 = hoffset >> 1;
231 vsync1 = voffset >> 1;
232 hoffset += hsync1;
233 voffset += vsync1;
234 config->HCycle = (int16_t)hcycle;
235 config->HSync0 = 0;
236 config->HSync1 = (int16_t)hsync1;
237 config->HOffset = (int16_t)hoffset;
238 config->VCycle = (int16_t)vcycle;
239 config->VSync0 = 0;
240 config->VSync1 = (int16_t)vsync1;
241 config->VOffset = (int16_t)voffset;
242
243 /* Verify */
244 eve_assert(config->HSync1);
245 eve_assert(config->HOffset > config->HSync1);
246 eve_assert(config->HCycle > config->Width);
247 eve_assert((config->HCycle - config->Width) > config->HOffset);
248 eve_assert(config->VSync1);
249 eve_assert(config->VOffset > config->VSync1);
250 eve_assert(config->VCycle > config->Height);
251 eve_assert((config->VCycle - config->Height) > config->VOffset);
252
253 return supportedResolution;
254}

344bool EVE_Util_bootup(EVE_HalContext *phost, EVE_BootupParameters *bootup)
345{
346 /* IMPORTANT: Do not use EVE_CoCmd functions here, as they can be overridden by hooks */
347
348 const uint32_t expectedChipId = EVE_CHIPID;
349 uint8_t engineStatus;
350 uint32_t chipId;
351 uint8_t id;
352 double syspll;
353 uint32_t freqbyte;
354 EVE_CMD_SYS_CLK_DIV sys_clk_div = { 0 };
355 EVE_CMD_BOOT_CFG boot_cfg = { 0 };
356 EVE_CMD_BOOT_CFG_EN boot_cfg_en = { 0 };
357 EVE_CMD_DDR_TYPE ddr_type = { 0 };
358
359 do
360 {
361 /* EVE will be in SPI Single channel after POR */
362 if (!EVE_Hal_powerCycle(phost, true))
363 return false;
364
365 boot_cfg_en.boot = 1;
366 boot_cfg_en.DDRtype = 1;
367 boot_cfg_en.GPREG = 0;
368 boot_cfg_en.enable = 1;
369 EVE_Hal_SPICmd_bootcfgen(phost, boot_cfg_en);
370 boot_cfg.DDR = 1;
371 boot_cfg.touch = 1;
372 boot_cfg.audio = 1;
373 boot_cfg.watchdog = 1;
374 boot_cfg.source = 0;
375 EVE_Hal_SPICmd_setbootcfg(phost, boot_cfg);
376 ddr_type.speed = 0; //1333
377 ddr_type.type = 1;
378 ddr_type.size = EVE_DDR_SIZE;
379 EVE_Hal_SPICmd_setddrtype(phost, ddr_type);
380 boot_cfg_en.boot = 1;
381 boot_cfg_en.DDRtype = 1;
382 boot_cfg_en.GPREG = 0;
383 boot_cfg_en.enable = 0;
384 EVE_Hal_SPICmd_bootcfgen(phost, boot_cfg_en); // enable(0)
385 syspll = 576000000;
386 freqbyte = (uint32_t)((round(syspll / FREQUENCY) - 1));
387 sys_clk_div.SYSPLL_CPS = 1; // default b'001'
388 sys_clk_div.freq = freqbyte; // system clock: 72MHz or 36MHz
389 EVE_Hal_SPICmd_sysclkdiv(phost, sys_clk_div);
390 EVE_Hal_SPICmd_pwr_state(phost, EVE_PWR_STATE_ACTIVE);
391 EVE_sleep(40);
392
393 while (EVE_Hal_rd32(phost, REG_BOOT_STATUS) != 0x522e2e2e) // Polling to indicate EVE is in normal running operation state
394 {
395 EVE_sleep(10);
396 eve_printf_debug("Boot status %lx \n", EVE_Hal_rd32(phost, REG_BOOT_STATUS));
397 }
398 eve_printf_debug("Boot status %lx \n", EVE_Hal_rd32(phost, REG_BOOT_STATUS));
399 chipId = EVE_Hal_rd32(phost, REG_CHIP_ID);
400 eve_printf_debug("chipID %lx \n", chipId);
401
402 while (EXTRACT_CHIPID(chipId) != EVE_BT820)
403 {
404 eve_printf_debug("EVE REG_CHIP_ID after wake up %lx\n", (unsigned long)chipId);
405
406 EVE_sleep(20);
407 if (phost->CbCmdWait && !phost->CbCmdWait(phost))
408 return false;
409
410 chipId = EVE_Hal_rd32(phost, REG_CHIP_ID);
411 }
412 } while (!chipId);
413
414 /* Validate chip ID to ensure the correct HAL is used */
415 /* ROM_CHIPID is valid across all EVE devices */
416 if (expectedChipId && EXTRACT_CHIPID(chipId) != expectedChipId)
417 eve_printf_debug("Mismatching EVE chip id %lx, expect model %lx\n", (unsigned long)((chipId >> 8) & 0xFF) | ((chipId & 0xFF) << 8), (unsigned long)expectedChipId);
418 eve_printf_debug("EVE chip id %lx %lx.%lx (EVE gen %i)\n", (unsigned long)EVE_shortChipId(EXTRACT_CHIPID(chipId)), (unsigned long)((chipId >> 16) & 0xFF), (unsigned long)((chipId >> 24) & 0xFF), EVE_gen(EXTRACT_CHIPID(chipId)));
419
420 /* Read Register ID to check if EVE is ready. */
421 while ((id = (uint8_t)EVE_Hal_rd32(phost, REG_ID)) != 0x7C)
422 {
423 eve_printf_debug("EVE register ID after wake up %x\n", (unsigned int)id);
424
425 EVE_sleep(20);
426 if (phost->CbCmdWait && !phost->CbCmdWait(phost))
427 return false;
428 }
429 eve_printf_debug("EVE register ID after wake up %x\n", (unsigned int)id);
430 eve_assert(chipId == EVE_Hal_rd32(phost, REG_CHIP_ID));
431
432
433 /* Read REG_CPURESET to check if engines are ready.
434 Bit 0 for coprocessor engine,
435 Bit 1 for touch engine,
436 Bit 2 for audio engine.
437 */
438 while ((engineStatus = (uint8_t)EVE_Hal_rd32(phost, REG_CPURESET)) != 0x00)
439 {
440 if (engineStatus & 0x01)
441 {
442 eve_printf_debug("Coprocessor engine is not ready\n");
443 }
444 if (engineStatus & 0x02)
445 {
446 eve_printf_debug("Touch engine is not ready\n");
447 }
448 if (engineStatus & 0x04)
449 {
450 eve_printf_debug("Audio engine is not ready\n");
451 }
452
453 EVE_sleep(20);
454 if (phost->CbCmdWait && !phost->CbCmdWait(phost))
455 return false;
456 }
457 eve_printf_debug("All engines are ready\n");
458
459 /* Update REG_FREQUENCY as per user selected */
460 if (bootup->SystemClock != EVE_SYSCLK_DEFAULT)
461 {
462 uint32_t clockMHz = bootup->SystemClock * 12;
463 EVE_Hal_wr32(phost, REG_FREQUENCY, clockMHz * 1000 * 1000);
464 EVE_Hal_flush(phost);
465 eve_printf_debug("EVE clock frequency set to %d MHz\n", (unsigned int)clockMHz);
466 }
467 else
468 {
469 eve_printf_debug("EVE default clock is %d MHz\n", (unsigned int)(EVE_Hal_rd32(phost, REG_FREQUENCY) / 1000000));
470 }
471
472 /* Switch to configured default SPI channel mode */
473 EVE_Hal_setSPI(phost, bootup->SpiChannels);
474#ifdef _DEBUG
475 const char *spiChannels = (phost->SpiChannels == EVE_SPI_QUAD_CHANNEL)
476 ? "Quad"
477 : ((phost->SpiChannels == EVE_SPI_DUAL_CHANNEL) ? "Dual" : "Single");
478 const char *requested = (bootup->SpiChannels == EVE_SPI_QUAD_CHANNEL)
479 ? "Quad"
480 : ((bootup->SpiChannels == EVE_SPI_DUAL_CHANNEL) ? "Dual" : "Single");
481 if (bootup->SpiChannels == phost->SpiChannels)
482 eve_printf_debug("%s channel SPI\n", spiChannels);
483 else
484 eve_printf_debug("%s channel SPI (%s unsupported)\n", spiChannels, requested);
485#endif
486
487 EVE_Hal_wr32(phost, REG_GPIO_DIR, 0xffffffff);
488 EVE_Hal_wr32(phost, REG_GPIO, 0xffffffff);
489
490 return true;
491}

501bool EVE_Util_config(EVE_HalContext *phost, EVE_ConfigParameters *config)
502{
503 /* IMPORTANT: Do not use EVE_CoCmd functions here, as they can be overridden by hooks */
504
505 bool reconfigure = phost->PCLK;
506 if (reconfigure)
507 {
508 /* If PCLK is already set, we're reconfiguring. Quiet flush first */
509 eve_printf_debug("Reconfiguring\n");
510 EVE_Cmd_waitFlush(phost);
511 }
512
513 uint32_t wp;
514 uint32_t rp;
515
516 eve_assert(config->Width < 2048);
517 eve_assert(config->Height < 2048);
518
519 phost->PCLK = 0;
520
521 EVE_Hal_wr32(phost, REG_HCYCLE, config->HCycle);
522 EVE_Hal_wr32(phost, REG_HOFFSET, config->HOffset);
523 EVE_Hal_wr32(phost, REG_HSYNC0, config->HSync0);
524 EVE_Hal_wr32(phost, REG_HSYNC1, config->HSync1);
525 EVE_Hal_wr32(phost, REG_VCYCLE, config->VCycle);
526 EVE_Hal_wr32(phost, REG_VOFFSET, config->VOffset);
527 EVE_Hal_wr32(phost, REG_VSYNC0, config->VSync0);
528 EVE_Hal_wr32(phost, REG_VSYNC1, config->VSync1);
529 EVE_Hal_wr32(phost, REG_PCLK_POL, config->PCLKPol);
530 EVE_Hal_wr32(phost, REG_HSIZE, config->Width);
531 EVE_Hal_wr32(phost, REG_VSIZE, config->Height);
532 EVE_Hal_wr32(phost, REG_DISP, 1);
533 EVE_Hal_wr32(phost, REG_PWM_DUTY, 128);
534
535#if defined(DISPLAY_RESOLUTION_WSVGA)
536 EVE_Hal_wr32(phost, REG_DISP, 1);
537 EVE_Hal_wr32(phost, REG_PCLK_POL, 1);
538
539 EVE_Hal_wr32(phost, REG_GPIO, 0x808b | (1 << 12));
540 EVE_Hal_wr32(phost, REG_DISP, 1);
541#endif
542
543 EVE_Util_clearScreen(phost);
544
545 /* Refresh fifo */
546 wp = EVE_Cmd_wp(phost);
547 rp = EVE_Cmd_rp(phost);
548 EVE_Cmd_space(phost);
549 phost->MediaFifoAddress = 0;
550 phost->MediaFifoSize = 0;
551
552 /* Coprocessor needs a reset */
553 if (reconfigure ? EVE_CMD_FAULT(rp) : (wp || rp))
554 {
555 /* If this occurs after powerdown and bootup, powerdown is not working */
556 eve_printf_debug("Coprocessor fifo not empty\n");
557 if (!EVE_Util_resetCoprocessor(phost))
558 return false;
559 }
560 else
561 {
562 /* Setup coprocessor defaults */
563 eve_printf_debug("Configure coprocessor defaults\n");
564
565 /* Reattach flash to avoid inconsistent state */
566 EVE_Cmd_wr32(phost, CMD_FLASHDETACH);
567 EVE_Cmd_wr32(phost, CMD_FLASHATTACH);
568 }
569
570
571 EVE_Hal_flush(phost);
572 if (!EVE_Cmd_waitFlush(phost))
573 return false;
574
575 phost->Width = config->Width;
576 phost->Height = config->Height;
577
578 if (EVE_DDR_SIZE == 0)
579 phost->DDR_RamSize = 128 * 1024 * 1024;
580 else if (EVE_DDR_SIZE == 2)
581 phost->DDR_RamSize = 512 * 1024 * 1024;
582 else
583 phost->DDR_RamSize = 128 * 1024 * 1024;
584
585#if 0
586 eve_printf_debug("REG_HSIZE: %i\n", (int)EVE_Hal_rd32(phost, REG_HSIZE));
587 eve_printf_debug("REG_VSIZE: %i\n", (int)EVE_Hal_rd32(phost, REG_VSIZE));
588 eve_printf_debug("REG_HCYCLE: %i\n", (int)EVE_Hal_rd32(phost, REG_HCYCLE));
589 eve_printf_debug("REG_HOFFSET: %i\n", (int)EVE_Hal_rd32(phost, REG_HOFFSET));
590 eve_printf_debug("REG_HSYNC0: %i\n", (int)EVE_Hal_rd32(phost, REG_HSYNC0));
591 eve_printf_debug("REG_HSYNC1: %i\n", (int)EVE_Hal_rd32(phost, REG_HSYNC1));
592 eve_printf_debug("REG_VCYCLE: %i\n", (int)EVE_Hal_rd32(phost, REG_VCYCLE));
593 eve_printf_debug("REG_VOFFSET: %i\n", (int)EVE_Hal_rd32(phost, REG_VOFFSET));
594 eve_printf_debug("REG_VSYNC0: %i\n", (int)EVE_Hal_rd32(phost, REG_VSYNC0));
595 eve_printf_debug("REG_VSYNC1: %i\n", (int)EVE_Hal_rd32(phost, REG_VSYNC1));
596 eve_printf_debug("REG_PCLK_POL: %i\n", (int)EVE_Hal_rd32(phost, REG_PCLK_POL));
597 eve_printf_debug("REG_DISP: %i\n", (int)EVE_Hal_rd32(phost, REG_DISP));
598 eve_printf_debug("REG_PWM_DUTY: %i\n", (int)EVE_Hal_rd32(phost, REG_PWM_DUTY));
599 eve_printf_debug("DDR_RamSize: %x\n", (int)phost->DDR_RamSize);
600#endif
601
602 eve_printf_debug("EVE configuration ready\n");
603 return true;
604}

656bool EVE_Util_resetCoprocessor(EVE_HalContext *phost)
657{
658 /* IMPORTANT: Do not use EVE_CoCmd functions here, as they can be overridden by hooks */
659 bool ready;
660 uint16_t rd;
661 uint16_t wr;
662 (void)rd;
663 (void)wr;
664
665 eve_printf_debug("Reset coprocessor\n");
666
667 /* Set REG_CPURESET to 1, to hold the coprocessor in the reset condition */
668 EVE_Hal_wr32(phost, REG_CPURESET, 1);
669 EVE_Hal_flush(phost);
670 EVE_sleep(100);
671
672 /* Set REG_CMD_READ and REG_CMD_WRITE to zero */
673 EVE_Hal_wr32(phost, REG_CMD_READ, 0);
674 EVE_Hal_wr32(phost, REG_CMD_WRITE, 0);
675 EVE_Hal_wr32(phost, REG_CMD_DL, 0);
676
677 /* Stop playing audio in case video with audio was playing during reset */
678 EVE_Hal_wr32(phost, REG_PLAYBACK_PLAY, 0);
679
680 /* Default */
681 phost->CmdFault = false;
682
683 /* Set REG_CPURESET to 0, to restart the coprocessor */
684 EVE_Hal_wr32(phost, REG_CPURESET, 0);
685 EVE_Hal_flush(phost);
686 EVE_sleep(100);
687 eve_assert((wr = EVE_Hal_rd32(phost, REG_CMD_WRITE)) == 0);
688 eve_assert((rd = EVE_Hal_rd32(phost, REG_CMD_READ)) == 0);
689
690 /* Refresh fifo */
691 EVE_Cmd_waitFlush(phost);
692 phost->MediaFifoAddress = 0;
693 phost->MediaFifoSize = 0;
694
695 /* Cold start. Ensure that the coprocessor is ready. */
696 EVE_Cmd_wr32(phost, CMD_DLSTART);
697 EVE_Cmd_wr32(phost, CMD_COLDSTART);
698
699 /* Reattach flash to avoid inconsistent state */
700 EVE_Cmd_wr32(phost, CMD_FLASHATTACH);
701
702 /* Wait for coprocessor to be ready */
703 ready = EVE_Cmd_waitFlush(phost);
704 if (phost->CbCoprocessorReset) /* Notify */
705 phost->CbCoprocessorReset(phost, !ready);
706
707#if defined(_DEBUG)
708 debugRestoreRamG(phost);
709#endif
710 return ready;
711}
712
718void EVE_Util_coprocessorFaultRecover(EVE_HalContext *phost)
719{
720 EVE_Hal_wr32(phost, REG_CMD_READ, 0);
721 while (EVE_Hal_rd32(phost, REG_CMD_WRITE) != 0)
722 ;
723
724 eve_assert((EVE_Hal_rd32(phost, REG_CMD_WRITE)) == 0);
725 eve_assert((EVE_Hal_rd32(phost, REG_CMD_READ)) == 0);
726
727 /* Default */
728 phost->CmdFault = false;
729
730#if defined(_DEBUG)
731 debugRestoreRamG(phost);
732#endif
733}

742bool EVE_Util_bootupConfig(EVE_HalContext *phost)
743{
744 EVE_BootupParameters bootup;
745 EVE_ConfigParameters config;
746
747 EVE_Util_bootupDefaults(phost, &bootup);
748 if (!EVE_Util_bootup(phost, &bootup))
749 {
750 return false;
751 }
752
753 EVE_Util_configDefaults(phost, &config, EVE_DISPLAY_DEFAULT);
754 if (!EVE_Util_config(phost, &config))
755 {
756 EVE_Util_shutdown(phost);
757 return false;
758 }
759
760 return true;
761}

769uint32_t EVE_Util_currentFrequency(EVE_HalContext *phost)
770{
771 uint32_t t0;
772 uint32_t t1;
773 uint32_t interval = 0;
774
775 t0 = EVE_Hal_rd32(phost, REG_CLOCK); /* t0 read */
776 /* may not be precise */
777 EVE_sleep(1000);
778
779 t1 = EVE_Hal_rd32(phost, REG_CLOCK); /* t1 read */
780
781 if (t1 > t0)
782 interval = t1 - t0;
783 else
784 interval = 0xFFFFFFFF - t0 + t1;
785 return (interval / 1000000);
786}
