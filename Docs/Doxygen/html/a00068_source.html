292static inline EVE_CHIPID_T EVE_extendedChipId(int chipId)
293{
294 switch (chipId & 0xFFFF)
295 {
296 case EVE_BT820 & 0xFFFF:
297 return (EVE_CHIPID_T)((chipId & 0xFFFF) | 0x50000);
298 default:
299 break;
300 }
301 return (EVE_CHIPID_T)(chipId & 0xFFFF);
302}
303
305static inline int EVE_shortChipId(EVE_CHIPID_T chipId)
306{
307 return chipId & 0xFFFF;
308}
309
310static inline int EVE_gen(EVE_CHIPID_T chipId)
311{
312 switch (chipId)
313 {
314 case EVE_BT820:
315 return EVE5;
316 default:
317 break;
318 }
319 return 0;
320}
