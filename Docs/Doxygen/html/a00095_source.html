44bool EVE_MediaFifo_set(EVE_HalContext *phost, uint32_t address, uint32_t size)
45{
46 bool res;
47
48 if (phost->MediaFifoAddress != address || phost->MediaFifoSize != size)
49 {
50 EVE_CoCmd_mediaFifo(phost, address, size);
51
52 /* Must flush for fifo pointers to be ready. */
53 res = EVE_Cmd_waitFlush(phost);
54 }
55 else
56 {
57 res = true;
58 }
59
60 if (res)
61 {
62 eve_assert(EVE_Hal_rd32(phost, REG_MEDIAFIFO_READ) < size);
63 eve_assert(EVE_Hal_rd32(phost, REG_MEDIAFIFO_WRITE) < size);
64 phost->MediaFifoAddress = address;
65 phost->MediaFifoSize = size;
66 }
67 else
68 {
69 phost->MediaFifoAddress = 0;
70 phost->MediaFifoSize = 0;
71 }
72
73 return res;
74}

115uint32_t EVE_MediaFifo_space(EVE_HalContext *phost)
116{
117 if (!phost->MediaFifoSize)
118 return 0;
119
120 int32_t rp = EVE_Hal_rd32(phost, REG_MEDIAFIFO_READ);
121 int32_t wp = EVE_Hal_rd32(phost, REG_MEDIAFIFO_WRITE);
122#if 1
123 return rp > wp
124 ? (rp - wp - 4)
125 : (rp + phost->MediaFifoSize - wp - 4);
126#else
127 int32_t diff = wp - rp;
128 if (diff < 0)
129 diff += phost->MediaFifoSize;
130 eve_assert(diff >= 0 && diff < phost->MediaFifoSize);
131 return phost->MediaFifoSize - diff - 4;
132#endif
133}

146bool EVE_MediaFifo_wrMem(EVE_HalContext *phost, const uint8_t *buffer, uint32_t size, uint32_t *transfered)
147{
148 if (!phost->MediaFifoSize)
149 {
150 eve_printf_debug("EVE Media FIFO has not been set, cannot write\n");
151 return false;
152 }
153
154#if 1
155 /* Two strategies.
156 - Wait for entire space and write the entire buffer.
157 - Wait for half the fifo to be available, and write in parts. */
158
159 /* If the size is within the media fifo size, write the entire buffer at once */
160 if (size <= phost->MediaFifoSize - 4)
161 {
162 uint32_t wp;
163 int32_t overflow;
164
165 if (!EVE_MediaFifo_waitSpace(phost, size, transfered))
166 return false;
167
168 wp = EVE_Hal_rd32(phost, REG_MEDIAFIFO_WRITE);
169 overflow = (int32_t)(wp + size) - (int32_t)(phost->MediaFifoSize);
170 if (overflow > 0)
171 {
172 eve_assert(phost->MediaFifoAddress + wp + size - overflow <= phost->DDR_RamSize);
173 eve_assert(phost->MediaFifoAddress + overflow <= phost->DDR_RamSize);
174 EVE_Hal_wrMem(phost, phost->MediaFifoAddress + wp, buffer, size - overflow);
175 EVE_Hal_wrMem(phost, phost->MediaFifoAddress, &buffer[size - overflow], overflow);
176 }
177 else
178 {
179 eve_assert(phost->MediaFifoAddress + wp + size <= phost->DDR_RamSize);
180 EVE_Hal_wrMem(phost, phost->MediaFifoAddress + wp, buffer, size);
181 }
182 wp += size;
183 if (wp >= phost->MediaFifoSize)
184 wp -= phost->MediaFifoSize;
185 eve_assert(wp < phost->MediaFifoSize);
186 EVE_Hal_wr32(phost, REG_MEDIAFIFO_WRITE, wp);
187 if (transfered)
188 *transfered = size;
189 return true;
190 }
191 else
192 {
193 /* Otherwise, write in parts. */
194 uint32_t halfSize = ((phost->MediaFifoSize >> 3) << 2) - 4;
195 uint32_t remaining = size;
196 uint32_t done = 0;
197 uint32_t wp;
198
199 wp = EVE_Hal_rd32(phost, REG_MEDIAFIFO_WRITE);
200 while (remaining)
201 {
202 uint32_t transfer = min(halfSize, remaining);
203 int32_t overflow;
204 if (!EVE_MediaFifo_waitSpace(phost, transfer, transfered))
205 return false;
206
207 overflow = (int32_t)(wp + transfer) - (int32_t)(phost->MediaFifoSize);
208 if (overflow > 0)
209 {
210 eve_assert(phost->MediaFifoAddress + wp + transfer - overflow <= phost->DDR_RamSize);
211 eve_assert(phost->MediaFifoAddress + overflow <= phost->DDR_RamSize);
212 EVE_Hal_wrMem(phost, phost->MediaFifoAddress + wp, &buffer[done], transfer - overflow);
213 EVE_Hal_wrMem(phost, phost->MediaFifoAddress, &buffer[done + transfer - overflow], overflow);
214 }
215 else
216 {
217 eve_assert(phost->MediaFifoAddress + wp + transfer <= phost->DDR_RamSize);
218 EVE_Hal_wrMem(phost, phost->MediaFifoAddress + wp, &buffer[done], transfer);
219 }
220 wp += transfer;
221 done += transfer;
222 remaining -= transfer;
223 if (wp >= phost->MediaFifoSize)
224 wp -= phost->MediaFifoSize;
225 eve_assert(wp < phost->MediaFifoSize);
226 EVE_Hal_wr32(phost, REG_MEDIAFIFO_WRITE, wp);
227
228 if (transfered)
229 {
230 *transfered = done;
231 if (EVE_Cmd_rp(phost) == EVE_Cmd_wp(phost))
232 return true; /* Early exit, finished processing. */
233 }
234 }
235
236 return true;
237 }
238#else
239 eve_scope()
240 {
241 uint32_t halfSize = ((phost->MediaFifoSize >> 3) << 2) - 4;
242 uint32_t remaining = size;
243 uint32_t done = 0;
244 uint32_t wp;
245
246 /* Write to media FIFO as soon as space is available */
247 wp = EVE_Hal_rd32(phost, REG_MEDIAFIFO_WRITE);
248 while (remaining)
249 {
250 uint32_t transfer = min(
251 EVE_MediaFifo_waitSpace(phost, 4),
252 min(halfSize, remaining));
253 if (!transfer)
254 return false; // !phost->CmdFault;
255 int32_t overflow = (int32_t)(wp + transfer) - (int32_t)(phost->MediaFifoSize);
256 if (overflow > 0)
257 {
258 EVE_Hal_wrMem(phost, phost->MediaFifoAddress + wp, &buffer[done], transfer - overflow);
259 EVE_Hal_wrMem(phost, phost->MediaFifoAddress, &buffer[done + transfer - overflow], overflow);
260 }
261 else
262 {
263 EVE_Hal_wrMem(phost, phost->MediaFifoAddress + wp, &buffer[done], transfer);
264 }
265 wp += transfer;
266 done += transfer;
267 remaining -= transfer;
268 if (wp >= phost->MediaFifoSize)
269 wp -= phost->MediaFifoSize;
270 eve_assert(wp < phost->MediaFifoSize);
271 EVE_Hal_wr32(phost, REG_MEDIAFIFO_WRITE, wp);

274 {
275 *transfered = done;
276 if (EVE_Cmd_rp(phost) == EVE_Cmd_wp(phost))
277 return true; /* Early exit, finished processing. */
278 }
279 }

298static bool checkWait(EVE_HalContext *phost, uint32_t rpOrSpace)
299{
300 if (EVE_CMD_FAULT(rpOrSpace))
301 {
302 char err[128];
303 (void)err;
304 /* Coprocessor fault */
305 phost->CmdWaiting = false;
306 eve_printf_debug("Coprocessor fault in processing media FIFO\n");
307#if defined(_DEBUG)
308 debugBackupRamG(phost);
309 EVE_Hal_rdMem(phost, (uint8_t *)err, RAM_REPORT, RAM_REPORT_MAX);
310 eve_printf_debug("%s\n", err);
311#endif
312 /* eve_debug_break(); */
313 return false;
314 }
315
316 if (phost->Status == EVE_STATUS_ERROR)
317 {
318 phost->CmdWaiting = false;
319 eve_printf_debug("Host error\n");
320 return false;
321 }
322
323 return true;
324}

334static bool handleWait(EVE_HalContext *phost, uint16_t rpOrSpace)
335{
336 /* Check for coprocessor fault */
337 if (!checkWait(phost, rpOrSpace))
338 return false;
339
340 /* Process any idling */
341 EVE_Hal_idle(phost);
342
343 /* Process user idling */
344 if (phost->CbCmdWait
345 && !phost->CbCmdWait(phost))
346 {
347 /* Wait aborted */
348 phost->CmdWaiting = false;
349 eve_printf_debug("Wait for media FIFO aborted\n");
350 return false;
351 }
352 return true;
353}

375uint32_t EVE_MediaFifo_waitSpace(EVE_HalContext *phost, uint32_t size, bool orCmdFlush)
376{
377 uint32_t space;
378
379 if (!phost->MediaFifoSize)
380 {
381 eve_printf_debug("EVE Media FIFO has not been set, cannot wait for space\n");
382 return 0;
383 }
384
385 if (size > (phost->MediaFifoSize - 4))
386 {
387 eve_printf_debug("Requested free space exceeds media FIFO\n");
388 return 0;
389 }
390
391 eve_assert(!phost->CmdWaiting);
392 phost->CmdWaiting = true;
393
394#if 1
395 space = EVE_MediaFifo_space(phost);
396 if (!checkWait(phost, space))
397 return 0;
398#endif
399
400 while (space < size)
401 {
402 space = EVE_MediaFifo_space(phost);
403 if (!handleWait(phost, (uint16_t)space))
404 return 0;
405 phost->CmdWaiting = false;
406 uint32_t cmdSpace = EVE_Cmd_waitSpace(phost, 0);
407 if (!cmdSpace)
408 return 0; /* Check for coprocessor error */
409 if (orCmdFlush && cmdSpace == (EVE_CMD_FIFO_SIZE - 4))
410 return 0; /* Processed */
411 phost->CmdWaiting = true;
412 }
413
414 phost->CmdWaiting = false;
415 return space;
416}
